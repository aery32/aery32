00175  * \param oscnum Oscillator number: 0, 1 or 32

00177 inline void aery_pm_wait_osc_to_stabilize(uint8_t oscnum)
00178 {
00179         switch (oscnum) {
00180         case 0:
00181                 while (!(AVR32_PM.isr & AVR32_PM_ISR_OSC0RDY_MASK));

00183         case 1:

00185                 break;
00186         case 32:
00187                 while (!(AVR32_PM.isr & AVR32_PM_ISR_OSC32RDY_MASK));
