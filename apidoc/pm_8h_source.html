112 /**
113  * Starts the chosen oscillator
114  *
115  * \note After start you shall wait OSC to stabilize
116  *
117  * \par Example
118  * \code
119  * // Start 12MHz crystal connected in XIN0/XOUT0
120  * aery_pm_start_osc(0, PM_OSC_MODE_GAIN3, PM_OSC_STARTUP_36ms);
121  * aery_pm_wait_osc_to_stabilize(0);
122  * \endcode
123  *
124  * \param oscnum Oscillator number which to start: 0, 1, 32
125  * \param mode Oscillator mode, see datasheet p. 74.
126  * \param startup Oscillator startup time
127  * \return Returns 0 on success and -1 on error
128  */

131 /**
132  * Initializes the frequency of PLL's voltage controlled oscillator (VCO)
133  *
134  * f_vco = f_src * mul / div, if div > 0
135  * f_vco = 2 * f_src * mul, if div == 0
136  *
137  * \note VCO frequency has to be in range of 80 MHz - 180 MHz or if hifreq mode
138  * is enabled in the range of 160 MHz - 240 MHz.
139  *
140  * \param ppll Pointer to pll register
141  * \param src Source oscillator
142  * \param mul Source frequency is multiplier by this value
143  * \param div Source frequency is divided by this value. If zero the
144  * multiplier is doubled
145  * \param hifreq Enable high frequency mode
146  */
147 int aery_pm_init_pllvco(volatile avr32_pm_pll_t *ppll, enum Pm_pll_source src, uint8_t mul,
148  uint8_t div, bool hifreq);
