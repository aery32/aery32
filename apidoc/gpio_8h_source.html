00173 /**
00174  * Enables GPIO local bus
00175  *
00176  * Before GPIO ports can be accessed through local bus you have to call
00177  * this function. Note that when local bus has been enabled you
00178  * cannot access GPIO ports through PBB (Peripheral Bus B) anymore, which
00179  * was the default bus accessing them, and thus functions such as
00180  * aery_gpio_set_pin_high() or aery_gpio_read_pin() has no effect. From now on
00181  * you have to use bitbanging and instead of using, for example, porta->ovr
00182  * you have to use new expression, lporta->ovr. To reroute GPIO back to
00183  * PBB call function gpio_disable_localbus().
00184  *
00185  * \attention CPU clock has to match with PBB clock to make local bus
00186  *            functional
00187  */
00188 inline void aery_gpio_enable_localbus(void)
00189 {
00190         __builtin_mtsr(AVR32_CPUCR,
00191                 __builtin_mfsr(AVR32_CPUCR) | AVR32_CPUCR_LOCEN_MASK);
00192 }
00193 
00194 /**
00195  * Disables GPIO local bus.
00196  *
00197  * \see gpio_enable_localbus()
00198  */ 
00199 inline void aery_gpio_disable_localbus(void)
00200 {
00201         __builtin_mtsr(AVR32_CPUCR,
00202                 __builtin_mfsr(AVR32_CPUCR) & ~AVR32_CPUCR_LOCEN_MASK);
