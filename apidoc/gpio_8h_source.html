55 /**

59  *
60  * \par Example:

133 /**

136  * \return Pin value high or low, 1 or 0
137  *
138  * \note Does not work for local bus
139  */
140 static inline bool aery_gpio_read_pin(uint8_t pinnum)
141 {
142  return (bool) AVR32_GPIO.port[GPIO_NUM2PORT(pinnum)].pvr
143  & (1UL << GPIO_NUM2PIN(pinnum));

146 /**
147  * Enables GPIO local bus
148  *
149  * Before GPIO ports can be accessed through local bus you have to call
150  * this function. Note that when local bus has been enabled you
151  * cannot access GPIO ports through PBB (Peripheral Bus B) anymore, which
152  * was the default bus accessing them, and thus functions such as
153  * aery_gpio_set_pin_high() or aery_gpio_read_pin() has no effect. From now on
154  * you have to use bitbanging and instead of using, for example, porta->ovr
155  * you have to use new expression, lporta->ovr. To reroute GPIO back to
156  * PBB call function gpio_disable_localbus().
157  *
158  * \attention CPU clock has to match with PBB clock to make local bus
159  * functional
160  */
161 static inline void aery_gpio_enable_localbus(void)
162 {
163  __builtin_mtsr(AVR32_CPUCR,
164  __builtin_mfsr(AVR32_CPUCR) | AVR32_CPUCR_LOCEN_MASK);
165 }
166 
167 /**
168  * Disables GPIO local bus.
169  *
170  * \see gpio_enable_localbus()
171  */
172 static inline void aery_gpio_disable_localbus(void)
173 {
174  __builtin_mtsr(AVR32_CPUCR,
175  __builtin_mfsr(AVR32_CPUCR) & ~AVR32_CPUCR_LOCEN_MASK);
176 }
