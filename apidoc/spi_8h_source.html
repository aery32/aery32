60 /**
61  * Setup SPI mode and the shift register width of the Numeric
62  * Processor Chip Select (NPCS, same as slave select). Every chip select
63  * line can have different mode and size of shift register.
64  *
65  * \par Making SPI SCK faster
66  * Chip select baudrate is hard coded to MCK/255. To make it faster
67  * you can bitbang the SCRB bit in CSRX register.
68  * \code
69  * aery_spi_setup_npcs(&AVR32_SPI0, 0, SPI_MODE1, 16);
70  * AVR32_SPI0.CSR0.scbr = 32; // baudrate is now MCK/32
71  * \endcode
72  *
73  * \param pspi Pointer to the SPI peripheral
74  * \param npcs Chip Select: 0-3
75  * \param mode SPI mode
76  * \param bits Width of SPI shift register: 8-16 bits
77  */
78 void aery_spi_setup_npcs(volatile avr32_spi_t *pspi, uint8_t npcs, enum Spi_mode mode, uint8_t bits);

80 /**
81  * Writes and reads SPI bus
82  *
83  * \par SPI read
84  * When you only want to read from external device through SPI, ignore the
85  * sent data and use dummy bits instead, for example 0x00.
86  * \code
87  * uint16_t rd; // read data
88  * rd = aery_spi_transmit(&AVR32_SPI0, 0, 0, true);
89  * \endcode
90  *
91  * \param pspi Pointer to the SPI peripheral which to use.
92  * \param data Binary word to be send.
93  * \param npcs Chip select line number. Can also be number of cs multiplexer.
94  * Use dummy cs number, if you like to use some other gpio pin
95  * through bitbanging.
96  * \param islast Is this the last transmit? If no, set 0 to leave chip select
97  * low.
98  * \return Received data bits
99  */
100 uint16_t aery_spi_transmit(volatile avr32_spi_t *pspi, uint16_t data, uint8_t npcs, bool islast);
